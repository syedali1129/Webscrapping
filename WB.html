# Selenium
def scrape_with_selenium():
    """Scrape using Selenium and return processing time."""
    start_time = time.time()
    driver = webdriver.Chrome()
    books = []

    for page_num in range(1, 6):
        url = f"https://www.amazon.in/gp/bestsellers/books/ref=zg_bs_pg_{page_num}"
        driver.get(url)
        book_elements = driver.find_elements(By.CSS_SELECTOR, '.zg-grid-general-faceout')

        for book in book_elements:
            try:
                title = book.find_element(By.CSS_SELECTOR, 'div._cDEzb_p13n-sc-css-line-clamp-1_1Fn1y').text
            except:
                title = "No Title"

            try:
                author = book.find_element(By.CSS_SELECTOR, 'a.a-size-small.a-link-child').text
            except:
                author = "No Author"

            try:
                price = book.find_element(By.CSS_SELECTOR, '.a-color-price').text
            except:
                price = "No Price"

            books.append({'Title': title, 'Author': author, 'Price': price})

    driver.quit()
    save_to_csv(books, '/Users/ali/Desktop/DATA 1202/webscrapping/selenium_books.csv')
    return time.time() - start_time


# Playwright
async def scrape_with_playwright():
    """Scrape using Playwright and return processing time."""
    start_time = time.time()
    books = []

    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        page = await browser.new_page()

        for page_num in range(1, 6):
            url = f"https://www.amazon.in/gp/bestsellers/books/ref=zg_bs_pg_{page_num}"
            await page.goto(url)
            await page.wait_for_load_state('networkidle')
            book_elements = await page.query_selector_all('.zg-grid-general-faceout')

            for book in book_elements:
                try:
                    title = await book.query_selector('div._cDEzb_p13n-sc-css-line-clamp-1_1Fn1y')
                    title = await title.inner_text() if title else "No Title"
                except:
                    title = "No Title"

                try:
                    author = await book.query_selector('a.a-size-small.a-link-child')
                    author = await author.inner_text() if author else "No Author"
                except:
                    author = "No Author"

                try:
                    price = await book.query_selector('.a-color-price')
                    price = await price.inner_text() if price else "No Price"
                except:
                    price = "No Price"

                books.append({'Title': title, 'Author': author, 'Price': price})

        await browser.close()
        save_to_csv(books, '/Users/ali/Desktop/DATA 1202/webscrapping/playwright_books.csv')
        return time.time() - start_time


# BeautifulSoup
def scrape_amazon_books(soup, books):
    start_time = time.time()
    book_elements = soup.find_all('div', class_='zg-grid-general-faceout')
    for book in book_elements:
        # Extract title
        title_element = book.find('div', class_='_cDEzb_p13n-sc-css-line-clamp-1_1Fn1y')
        title = title_element.get_text(strip=True) if title_element else 'N/A'
        
        # Extract author
        author = book.find('a', class_='a-size-small a-link-child').get_text(strip=True) if book.find('a', class_='a-size-small a-link-child') else 'N/A'
        
        # Extract price
        price_element = book.find('span', class_='a-color-price')
        price = price_element.get_text(strip=True) if price_element else 'N/A'
        
        # Extract rating
        rating_element = book.find('span', class_='a-icon-alt')
        rating = rating_element.get_text(strip=True) if rating_element else 'N/A'
        
        # Extract number of customer ratings
        customer_rated_element = book.find('span', class_='a-size-small')
        customer_rated = customer_rated_element.get_text(strip=True) if customer_rated_element else 'N/A'
        
        # Append to the books list
        books.append({
            'Title': title,
            'Author': author,
            'Price': price,
            'Rating': rating,
            'Customer_Rated': customer_rated
        })
    return time.time() - start_time
